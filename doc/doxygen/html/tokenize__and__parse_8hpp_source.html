74 using result_type = tokenize_and_parse_result<Source, TokenizerParseContext, ParserParseContext>;
75
76 std::shared_ptr<result_type> result = std::make_shared<result_type>();
77
78 //tokenize
79 result->tokenizer.parse_context = TokenizerParseContext(source);
80 result->tokenizer.success = result->tokenizer.parse_context.parse(tokenizer_grammar) && result->tokenizer.parse_context.errors().empty();
81
82 //parse
83 result->parser.parse_context = result->tokenizer.parse_context.template derive_parse_context<typename ParserParseContext::match_id_type, typename ParserParseContext::error_id_type>();
84 result->parser.success = result->parser.parse_context.parse(parser_grammar) && result->parser.parse_context.errors().empty();
85
86 //make ast
87 result->ast = make_ast(result->parser.parse_context.matches(), astFactory);
88
89 //total success
90 result->success = result->tokenizer.success && result->parser.success;
91
92 //common errors
93 if constexpr (result_type::same_error_id_type) {
94 //add the tokenizer errors
95 for (const auto& tokenizerError : result->tokenizer.parse_context.errors()) {
96 result->errors.push_back(tokenizerError);
97 }
98
99 //add the parser errors
100 for (const auto& parserError : result->parser.parse_context.errors()) {
101 result->errors.push_back(parse_error<Source, typename TokenizerParseContext::error_id_type, typename TokenizerParseContext::text_position_type>(parserError.id(), parserError.begin_iterator()->start_position(), parserError.end_iterator()->begin_iterator()));
102 }
103
104 //sort the errors by starting position
105 std::sort(result->errors.begin(), result->errors.end(), [](const auto& errorA, const auto& errorB) {
106 return errorA.begin() < errorB.begin();
107 });
108 }
