209 bool parse(ParseContext& pc) const {
210 const auto it = pc.m_rule_data.find(this);
211
212 //initial entrance
213 if (it == pc.m_rule_data.end()) {
214 const rule_data_type initial_state = { pc.parse_position().iterator(), ParseContext::rule_state::none };
215 auto [it1, ok] = pc.m_rule_data.insert(std::make_pair(this, initial_state));
216 return _parse(pc, it1->second, initial_state);
217 }
218
219 //subsequent entrance; start new state if iterator has advanced from last time
220 else if (it->second.iterator != pc.parse_position().iterator()) {
221 const auto prev_state = it->second;
222 it->second.iterator = pc.parse_position().iterator();
223 it->second.state = ParseContext::rule_state::none;
224 return _parse(pc, it->second, prev_state);
225 }
226
227 //left recursion found
228 else {
229 switch (it->second.state) {
230 case ParseContext::rule_state::none:
231 throw left_recursion{ this };
232
233 case ParseContext::rule_state::reject:
234 return false;
235
236 case ParseContext::rule_state::accept:
237 pc.m_terminal_parsing_allowed = true;
238 return true;
239 }
240 }
241
242 throw std::logic_error("rule: parse: invalid state");
243 }

312 //parse rule; on exception, restore rule data
313 bool _parse(ParseContext& pc, rule_data_type& rd, const rule_data_type& prev_rd) const {
314 try {
315 const bool result = m_parse_node->parse(pc);
316 rd = prev_rd;
317 return result;
318 }
319 catch (left_recursion ex) {
320 if (ex.rule == this) {
321 try {
322 const bool result = parse_left_recursion(pc, rd);
323 rd = prev_rd;
324 return result;
325 }
326 catch (left_recursion ex) {
327 rd = prev_rd;
328 throw infinite_recursion_exception( ex.rule->text(), pc.parse_position().to_string(pc.begin()) );
329 }
330 catch (...) {
331 rd = prev_rd;
332 throw;
333 }
334 }
335 else {
336 rd = prev_rd;
337 throw ex;
338 }
339 }
340 }

342 //parse left recursion
343 bool parse_left_recursion(ParseContext& pc, rule_data_type& rd) const {
344 const auto prev_left_recursion_start_state = pc.m_left_recursion_start_state;
345 const bool prev_terminal_parsing_allowed = pc.m_terminal_parsing_allowed;
346
347 pc.m_left_recursion_start_state = pc.get_state();
348
349 //reject
350 rd.state = ParseContext::rule_state::reject;
351 if (!m_parse_node->parse(pc)) {
352 return false;
353 }
354
355 //accept
356 rd.state = ParseContext::rule_state::accept;
357 for(;;) {
358 rd.iterator = pc.m_parse_position.iterator();
359 pc.m_terminal_parsing_allowed = false;
360 try {
361 if (!m_parse_node->parse(pc)) {
362 break;
363 }
364 }
365 catch (...) {
366 pc.m_left_recursion_start_state = prev_left_recursion_start_state;
367 pc.m_terminal_parsing_allowed = prev_terminal_parsing_allowed;
368 throw;
369 }
370 }
371
372 //success
373 pc.m_left_recursion_start_state = prev_left_recursion_start_state;
374 pc.m_terminal_parsing_allowed = prev_terminal_parsing_allowed;
375 return true;
376 }
