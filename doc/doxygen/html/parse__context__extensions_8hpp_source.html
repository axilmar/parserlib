157 template <class ParseContext, class ParseNode, class Annotation>
158 bool parse_annotation(ParseContext& pc, const ParseNode& parse_node, const Annotation& annotation) {
159 if (m_output_stream) {
160 *m_output_stream << std::string(m_indentation_level * m_indentation_size, ' ') << "Parsing " << annotation << " at " << pc.parse_position().to_string(pc.begin_iterator()) << std::endl;
161 ++m_indentation_level;
162 }
163 const bool result = pc.parse(parse_node);
164 if (m_output_stream) {
165 --m_indentation_level;
166 if (result) {
167 *m_output_stream << std::string(m_indentation_level * m_indentation_size, ' ') << "Successfully parsed " << annotation << std::endl;
168 }
169 else {
170 *m_output_stream << std::string(m_indentation_level * m_indentation_size, ' ') << "Failed to parse " << annotation << std::endl;
171 }
172 }
173 return result;
174 }
