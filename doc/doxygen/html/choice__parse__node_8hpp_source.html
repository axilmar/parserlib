94 template <class ParseContext, class ErrorState, class ErrorContainer>
95 static bool get_branch_errors(ParseContext& pc, const ErrorState& start_error_state, ErrorContainer& branch_errors) {
96 if (pc.errors().size() > start_error_state.error_count()) {
97 if (!branch_errors.empty()) {
98 const auto& last_error = pc.errors().back();
99 const auto& last_branch_error = branch_errors.back();
100 if (last_error.begin() > last_branch_error.begin()) {
101 branch_errors = pc.errors(start_error_state);
102 }
103 }
104 else {
105 branch_errors = pc.errors(start_error_state);
106 }
107 return true;
108 }
109 return false;
110 }

112 template <class ParseNode, class ParseContext, class State, class ErrorState, class ErrorContainer>
113 static bool invoke_child(const ParseNode& child, ParseContext& pc, const State& start_state, const ErrorState& start_error_state, State& branch_state, ErrorContainer& branch_errors, bool& branch_result) {
114 //if parse is successful
115 if (pc.parse(child)) {
116 //if parsed without new errors, then stop branch testing and return true;
117 //keep the errors
118 if (!get_branch_errors(pc, start_error_state, branch_errors)) {
119 pc.add_errors(branch_errors);
120 return true;
121 }
122
123 //continue branch testing to get the longest parsing possible;
124 //since this branch returned true, save the parse state to return it later
125 branch_state = pc.get_state();
126 branch_result = true;
127 }
128
129 //else failed to parse; collect the branch errors
130 else {
131 get_branch_errors(pc, start_error_state, branch_errors);
132 }
133
134 //restore the state for the next branch
135 pc.set_state(start_state);
136 pc.set_error_state(start_error_state);
137 return false;
138 }

140 template <size_t Index, class ParseContext, class State, class ErrorState, class ErrorContainer>
141 bool _parse(ParseContext& pc, const State& start_state, const ErrorState& start_error_state, State& branch_state, ErrorContainer& branch_errors, bool& branch_result) const {
142 if constexpr (Index < std::tuple_size_v<std::tuple<ParseNodes...>>) {
143 const auto& child = std::get<Index>(m_children);
144 if (invoke_child(child, pc, start_state, start_error_state, branch_state, branch_errors, branch_result)) {
145 return true;
146 }
147 return _parse<Index + 1>(pc, start_state, start_error_state, branch_state, branch_errors, branch_result);
148 }
149 else {
150 pc.add_errors(branch_errors);
151 pc.set_state(branch_state);
152 return branch_result;
153 }
154 }
