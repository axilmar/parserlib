557 template <class ParseNode, std::enable_if_t<std::is_base_of_v<parse_node_base, std::decay_t<ParseNode>>, bool> = true>
558 bool parse_memoized(const ParseNode& parse_node, int id) {
559 //the key is the combination of the parse position and parse node address
560 const auto key = std::make_pair(m_parse_position.iterator(), id);
561
562 //find the key
563 const auto it = m_memoized.find(key);
564
565 //if found, restore memoized state
566 if (it != m_memoized.end()) {
567 const auto& mem = it->second;
568 m_parse_position = std::get<1>(mem);
569 m_matches.insert(m_matches.end(), std::get<2>(mem).begin(), std::get<2>(mem).end());
570 m_errors.insert(m_errors.end(), std::get<3>(mem).begin(), std::get<3>(mem).end());
571 return std::get<0>(mem);
572 }
573
574 //else parse and memoize state
575 const auto start_state = get_state();
576 const auto start_error_state = get_error_state();
577 const bool result = parse(parse_node);
578 m_memoized.insert(std::make_pair(key, std::make_tuple(result, m_parse_position, match_container_type(std::next(m_matches.begin(), start_state.match_count()), std::next(m_matches.begin(), m_matches.size())), error_container_type(std::next(m_errors.begin(), start_error_state.error_count()), std::next(m_errors.begin(), m_errors.size())))));
579 return result;
580 }
