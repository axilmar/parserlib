26 template <class ParseContext, class ParseNode>
27 static bool parse(ParseContext& pc, const ParseNode& pn) {
28 //get the left recursion state for the given parse node
29 auto& lrs = pc.get_left_recursion_state(std::addressof(pn));
30
31 //if the parse iterator has advanced from the last time, enter a new no-left-recursion state
32 if (pc.get_iterator() != lrs.get_iterator() || lrs.is_initial_state()) {
33 return parse_no_left_recursion(pc, pn, lrs);
34 }
35
36 //since the iterator kept for the given parse node remains the same as the
37 //current iterator of the parse context, we have a left recursion;
38 //handle it according to status
39 switch (lrs.get_status()) {
40 //current status is no left recursion, therefore initiate left recursion handling;
41 //throw an exception instead of handling it here in order to avoid failure
42 //due to following parse nodes that must parse successfully.
43 case left_recursion_status::no_left_recursion:
44 throw left_recursion_exception(std::addressof(pn));
45
46 //reject the left recursion to allow non-left recursive parts of the grammar to parse.
47 case left_recursion_status::reject_left_recursion:
48 return false;
49
50 //accept the left recursion without parsing, and unblock parsing for terminals.
51 case left_recursion_status::accept_left_recursion:
52 pc.unblock_parsing();
53 return true;
54 }
55
56 //if execution reaches this point, then there is memory corruption,
57 //because the status value would be invalid
58 throw std::runtime_error("parse_left_recursion_algorithm: parse_left_recursion: memory corruption: invalid left recursion status.");
59 }

63 template <class ParseContext, class ParseNode, class LeftRecursionState>
64 static bool parse_left_recursion_status(ParseContext& pc, const ParseNode& pn, LeftRecursionState& lrs, left_recursion_status status) {
65 //keep current state in order to restore it later
66 const LeftRecursionState prev_state = lrs;
67
68 //set new state
69 lrs = LeftRecursionState(pc.get_iterator(), status);
70
71 //try to parse
72 try {
73 const bool result = pn.parse(pc);
74
75 //restore the state
76 lrs = prev_state;
77
78 //return the parse result
79 return result;
80 }
81
82 //on exception, restore the parse state
83 catch (...) {
84 lrs = prev_state;
85 throw;
86 }
87 }

92 static bool parse_no_left_recursion(ParseContext& pc, const ParseNode& pn, LeftRecursionState& lrs) {

104 template <class ParseContext, class ParseNode, class LeftRecursionState>
105 static bool handle_left_recursion(ParseContext& pc, const ParseNode& pn, LeftRecursionState& lrs) {
106 //get the match parse state to later activate it for the accept phase,
107 //in order to maintain left associativity.
108 const auto mps = pc.get_match_parse_state();
109
110 //try the reject phase
111 if (!parse_left_recursion_status(pc, pn, lrs, left_recursion_status::reject_left_recursion)) {
112 return false;
113 }
114
115 //try the accept phase; loop until no more parsing can be done
116 for (;;) {
117 //activate the match parse state, so as that following matches
118 //start from the correct point
119 pc.set_match_parse_state(mps);
120
121 //block terminal parsing; before any terminal is parsed,
122 //there must be an 'accept'.
123 pc.block_parsing();
124
125 //parse the accept phase
126 if (!parse_left_recursion_status(pc, pn, lrs, left_recursion_status::accept_left_recursion)) {
127 pc.unblock_parsing();
128 break;
129 }
130 }
131
132 //successfully parsed a left recursion
133 return true;
134 }
